create table public.agency_settings (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone not null default timezone ('utc'::text, now()),
  agency_name text not null,
  subdomain text not null,
  package_level text null default 'bronze'::text,
  logo_url text null,
  font_family text null default 'Inter, sans-serif'::text,
  primary_color text null default '#10b981'::text,
  hero_type text null default 'image'::text,
  hero_url text null,
  hero_title text null,
  modules_active jsonb null default '{"contact_page": true, "project_tracker": false}'::jsonb,
  footer_config jsonb null default '{"email": "contact@agency.com", "phone": "+34 000 00 00 00", "address": "Calle Mayor, Alicante", "socials": {"linkedin": "", "instagram": ""}}'::jsonb,
  filter_config jsonb null default '{"minCommission": 0, "excludedPromoters": []}'::jsonb,
  constraint agency_settings_pkey primary key (id),
  constraint agency_settings_subdomain_key unique (subdomain),
  constraint agency_settings_hero_type_check check (
    (
      hero_type = any (array['image'::text, 'video'::text])
    )
  ),
  constraint agency_settings_package_level_check check (
    (
      package_level = any (
        array['bronze'::text, 'silver'::text, 'gold'::text]
      )
    )
  )
) TABLESPACE pg_default;
